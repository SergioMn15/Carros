async function cargarReportes() { const tabla = document.getElementById('tabla-reportes'); try { const response = await fetch('/api/reporte'); const datos = await response.json(); tabla.innerHTML = datos.map(r => ` ${new Date(r.fecha).toLocaleString()} ${r.placa} ${r.falla} ${r.ubicacion} ✅ Recibido `).join(''); } catch (e) { tabla.innerHTML = "Error al cargar datos"; } } cargarReportes();
